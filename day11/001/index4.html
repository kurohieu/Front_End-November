Câu lệnh rẽ nhánh switch
        cú pháp
        switch(biểu thức) {
            case label1:
                // các câu lệnh thực hiện nếu biểu thức bằng label1
                break;
            case label2:
                // các câu lệnh thực hiện nếu biểu thức bằng label2
                break;
            case label3:
                // các câu lệnh thực hiện nếu biểu thức bằng label3
                break;
            ...
            default:
                // các câu lệnh được chạy nếu biểu thức khác tất cả nhãn được
                // khai báo label1, label2 , label3 ...

        }

        thông thường biểu thức trong () sau switch nó là 1 biến
